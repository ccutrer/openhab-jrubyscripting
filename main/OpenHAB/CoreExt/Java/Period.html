def to_i
  [java.time.temporal.ChronoUnit::YEARS,
   java.time.temporal.ChronoUnit::MONTHS,
   java.time.temporal.ChronoUnit::DAYS].sum do |unit|
    get(unit) * unit.duration.to_i
  end
end
