  ([self] +
    Array(superclass&.ancestors) +
    interfaces.flat_map(&:ancestors)).uniq
end

def generic_ancestors
  ancestors.flat_map do |klass|
    Array(klass.generic_superclass) + klass.generic_interfaces
  end.uniq
end
